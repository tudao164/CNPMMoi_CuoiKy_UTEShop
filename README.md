      this.token = data.data.token;
      localStorage.setItem('token', this.token);
    }
    return data;
  }

  async getProfile() {
    const response = await fetch(`${this.baseURL}/api/user/profile`, {
      headers: { 'Authorization': `Bearer ${this.token}` }
    });
    return response.json();
  }
}
